document.addEventListener('DOMContentLoaded', function() {
    var elm = document.querySelectorAll(".thumbnail, .subcategory");
    var show_block = document.querySelector(".block_show_more");
    var width_window = window.outerWidth;

    if (width_window >= 991){
        if (elm.length > 12) {
            show_block.classList.add('hidden_show_block');
            let show_more = document.createElement('a');
            show_more.classList.add('show_more_block_button');
            show_more.innerHTML = 'Все категории';
            show_block.after(show_more);
            document.addEventListener("click", function(e) {
                    if (e.target.className=="show_more_block_button") {
                        show_block.classList.toggle('hidden_show_block');
                        show_more.innerHTML == 'Все категории' ? show_more.innerHTML = 'Скрыть категории' : show_more.innerHTML = 'Все категории';
                    }
            });
        }
    } else {
        if (elm.length > 4) {
            show_block.classList.add('hidden_show_block');

            show_more.classList.add('show_more_block_button');

    }
